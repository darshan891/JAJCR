import os
import json
import getpass
import wikipediaapi

from pydantic import BaseModel
from langchain_cohere import ChatCohere
from langchain_core.prompts import ChatPromptTemplate

# =========================================
# ENTER COHERE API KEY
# =========================================

os.environ["COHERE_API_KEY"] = getpass.getpass("Enter Cohere API Key: ")

# =========================================
# LOAD COHERE MODEL
# =========================================

model = ChatCohere(
    model="command-r7b-12-2024"
)

# =========================================
# PYDANTIC SCHEMA
# =========================================

class InstitutionInfo(BaseModel):
    name: str
    founder: str
    founded_year: str
    branches: str
    employees: str
    summary: str

# =========================================
# FETCH DATA FROM WIKIPEDIA
# =========================================

def fetch_wikipedia_summary(institution_name):

    wiki = wikipediaapi.Wikipedia(
        user_agent='my-app',
        language='en'
    )

    page = wiki.page(institution_name)

    if not page.exists():
        return None

    # Limit text size for better output
    return page.summary[:2000]

# =========================================
# EXTRACTION FUNCTION
# =========================================

def extract_info(institution_name):

    text = fetch_wikipedia_summary(institution_name)

    if text is None:
        return "Wikipedia page not found"

    prompt = ChatPromptTemplate.from_template("""
Extract structured information from the text.

Text:
{text}

Return ONLY valid JSON in this format:

{{
    "founder": "",
    "founded_year": "",
    "branches": "",
    "employees": "",
    "summary": ""
}}
""")

    # Create chain
    chain = prompt | model

    # Invoke model
    response = chain.invoke({"text": text}).content

    print("Raw LLM Output:\n")
    print(response)

    # =========================================
    # CLEAN RESPONSE
    # =========================================

    cleaned_response = response.replace("```json", "") \
                               .replace("```", "") \
                               .strip()

    # =========================================
    # CONVERT JSON STRING TO PYTHON DICTIONARY
    # =========================================

    try:
        data = json.loads(cleaned_response)

        result = InstitutionInfo(
            name=institution_name,
            founder=data.get("founder", ""),
            founded_year=data.get("founded_year", ""),
            branches=data.get("branches", ""),
            employees=data.get("employees", ""),
            summary=data.get("summary", "")
        )

        return result

    except Exception as e:
        return f"Error parsing JSON: {e}"

# =========================================
# RUN PROGRAM
# =========================================

result = extract_info("Google")

print("\n========== FINAL OUTPUT ==========\n")
print(result)
