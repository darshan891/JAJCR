from gensim.models import Word2Vec
from nltk.tokenize import word_tokenize
import nltk

nltk.download('punkt')

# Sample corpus
corpus = [
    "Artificial intelligence is transforming technology",
    "Machine learning improves automation",
    "Deep learning is a subset of AI",
    "AI helps in healthcare and education",
    "Technology is evolving rapidly"
]

# Tokenization
tokenized_corpus = [word_tokenize(sentence.lower()) for sentence in corpus]

# Train Word2Vec model
model = Word2Vec(sentences=tokenized_corpus, vector_size=100,
window=5, min_count=1, workers=4)

# Function to get similar words
def get_similar_words(word):
    return model.wv.most_similar(word, topn=5)

# Original prompt
prompt = "AI improves healthcare"

print("Original Prompt:")
print(prompt)

# Enrich prompt using similar words
similar_words = get_similar_words("AI")

enriched_prompt = prompt + " using " + ", ".join([word for word, score in similar_words])

print("\nEnriched Prompt:")
print(enriched_prompt)
