import gensim.downloader as api
import numpy as np

# Load pre-trained Word2Vec model
model = api.load("word2vec-google-news-300")

# Words for analysis
word1 = "king"
word2 = "queen"
word3 = "man"
word4 = "woman"

# Find similarity between words
similarity = model.similarity(word1, word2)

print(f"Similarity between {word1} and {word2}: ", similarity)

# Vector arithmetic: king - man + woman ≈ queen
result = model.most_similar(positive=[word1, word4], negative=[word3], topn=1)

print("\nVector Arithmetic Result:")
print(f"{word1} - {word3} + {word4} = {result[0][0]}")

# Find similar words
similar_words = model.most_similar(word1, topn=5)

print(f"\nTop 5 similar words to '{word1}':")
for word, score in similar_words:
    print(word, ":", score)
